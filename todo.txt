/**
 * Deregister WP Embed script.
 */
function greenlet_deregister_scripts() {
	$disable = gl_get_option( 'disable_embed', false );
	if ( false === $disable ) {
		return;
	}

	wp_deregister_script( 'wp-embed' );
}

add_action( 'wp_footer', 'greenlet_deregister_scripts' );
